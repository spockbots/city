from spockbots.motor import SpockbotsMotor
import time


[docs]def run_calibrate():
    """
    Run the calibration

    :return: a file called calibrate.txt that containes the minimum black and the maximum white value for the sensors
    """
    robot = SpockbotsMotor()
    robot.debug = True

    robot.setup()

    print(robot)

    robot.beep()
    robot.calibrate(10, distance=30, ports=[2, 3, 4], direction='front')
    robot.beep()

    robot.color.write()
