[docs]    def value(self):
        """
        reads the current value mapped between 0 and 100
        :return:
        """
        val = self.sensor.reflection()
        b = self.black
        t1 = val - b
        t2 = self.white - self.black
        ratio = t1 / t2
        c = ratio * 100
        if c < 0:
            c = 0
        if c > 100:
            c = 100
        return int(c)
